    const todo = ["Breakfast", "Check a Email", " Start Working(Coloring)",

    let x = 0;
    while (x < todo.length){
        let todoLi = document.createElement('li');
        todoLi.innerHTML = todo[x];
        x++;   
        document.getElementById("list-todo").appendChild(todoLi);  
    }
</script>
